\section[方程$Z(n)=\varphi_{e}(SL(n))$的可解性]{方程$Z(n)=\varphi_{e}(SL(n))$的可解性}

\begin{frame}
\hilite<1> \frametitle{方程$Z(n)=\varphi_{e}(SL(n))$的可解性} \pause

\begin{itemize}
\hilite<2> \item 拟研究内容: \pause

\begin{itemize}
\hilite<3>\item 当$e\in\{1,2,3,4,6\}$或$e\mid\varphi(SL(n))(e>1)$时, 方程$Z(n)=\varphi_{e}(SL(n))$的可解性？\pause
\hilite<4> \item 当$e\in\{p,p^{2}, pq,p^{t}\}$时, 方程$Z(n)=\varphi_{e}(SL(n))$的可解性？\pause
\hilite<5> \item 对于任意的正整数$e$, 能否得到方程$Z(n)=\varphi_{e}(SL(n))$的全部正整数解? \pause
\end{itemize}
\end{itemize}

\begin{itemize}
\hilite<6> \item 拟研究方法: \pause

\begin{itemize}
\hilite<7> \item 基于广义欧拉函数$\varphi_{e}(n)$的计算公式, 根据Smarandache LCM函数的性质, 将正整数$n$的标准分解式, 按其偶素数方幂和奇素数方幂的大小进行分类.
 \pause
\hilite<8> \item 利用伪Smanrandache函数的定义, 得到整除关系式, 借助初等的方法和技巧进一步计算整除问题.
\end{itemize}
\end{itemize}
\end{frame}
